# Game Store

The game store is the single source of truth for runtime state: entities, party, inventory, world flags, and animation state. Built with Pinia's composition API.

## Overview

```ts
// stores/game.ts
export const useGameStore = defineStore('game', () => {
  // World
  const currentScene = ref<string>('main')
  const worldFlags = ref<Record<string, boolean | number>>({})

  // Entities
  const entities = ref<Map<string, EntityState>>(new Map())

  // Party & Inventory
  const party = reactive({
    members: [] as string[],
    leader: null as string | null,
    inventory: [] as Item[],
  })

  // Animation (for UI controllers)
  const animations = ref<string[]>([])
  const currentAnimation = ref<string | null>(null)
})
```

## State Domains

| Domain | Purpose |
|--------|---------|
| **entities** | All spawned game objects (characters, items, interactables) |
| **party** | Player-controlled characters and their leader |
| **inventory** | Items the party owns |
| **worldFlags** | Quest progress, unlocked areas, game state |
| **animations** | Available animations for UI display |

## Entity State

```ts
export interface Position {
  x: number
  y: number
  z: number
}

export interface EntityState {
  id: string              // Unique instance ID
  templateId: string      // Reference to YAML template
  type: 'character' | 'item' | 'interactable'
  name: string
  position: Position
  rotation?: Position

  // Rendering
  model?: string
  animations?: Record<string, string>

  // Character-specific
  faction?: string
  hostile?: boolean
  controllable?: boolean
  hp?: number
  maxHp?: number
  stats?: Record<string, number>
  ai?: { behavior: string; [key: string]: unknown }

  // Interactable-specific
  locked?: boolean
  opened?: boolean
  destructible?: boolean

  // Equipment
  equipment?: Equipment
}

export interface Equipment {
  mainHand?: string  // weapon templateId
  offHand?: string   // weapon templateId
}
```

## Scene Actions

### Load Scene

Loads a scene from YAML, clears world entities (keeps party), spawns scene entities:

```ts
async function loadScene(sceneId: string, spawnPointId: string = 'default') {
  const scene = await queryCollection('scenes')
    .where('sceneId', '=', sceneId)
    .first()

  if (!scene) {
    throw new Error(`Scene not found: ${sceneId}`)
  }

  // Clear only NON-party entities
  for (const [id] of entities.value) {
    if (!party.members.includes(id)) {
      entities.value.delete(id)
    }
  }

  currentScene.value = sceneId

  // Spawn scene entities
  for (const entry of scene.entities) {
    await spawnFromTemplate(
      entry.templateId,
      entry.position,
      {
        rotation: entry.rotation,
        ...entry.overrides,
      },
    )
  }

  const spawnPoint = scene.spawnPoints[spawnPointId]
    ?? scene.spawnPoints.default

  return { scene, spawnPoint }
}
```

Usage:
```ts
// Load scene, position party at spawn point
const { spawnPoint } = await gameStore.loadScene('goblin_hideout', 'from_forest')
gameStore.updateEntity(partyLeaderId, { position: spawnPoint })
```

See [Scenes](/entities/scenes) for scene YAML format.

## Entity Actions

### Spawn from Template

Creates runtime instance from YAML template:

```ts
async function spawnFromTemplate(
  templateId: string,
  position: Position,
  overrides?: Partial<EntityState>,
): Promise<string> {
  const template = await queryCollection('entities')
    .where('templateId', '=', templateId)
    .first()

  if (!template) {
    throw new Error(`Template not found: ${templateId}`)
  }

  const instanceId = `${template.templateId}_${Date.now()}`

  const entityState: EntityState = {
    id: instanceId,
    templateId: template.templateId,
    type: template.type,
    name: template.name,
    position,
    model: template.model,
    animations: template.animations,
    faction: template.faction,
    hostile: template.hostile,
    controllable: template.controllable,
    hp: template.stats?.hp,
    maxHp: template.stats?.maxHp,
    stats: template.stats,
    ai: template.ai,
    ...overrides,
  }

  spawnEntity(instanceId, entityState)
  return instanceId
}
```

### CRUD Operations

```ts
function spawnEntity(id: string, state: EntityState) {
  entities.value.set(id, state)
}

function updateEntity(id: string, partial: Partial<EntityState>) {
  const entity = entities.value.get(id)
  if (entity) {
    entities.value.set(id, { ...entity, ...partial })
  }
}

function removeEntity(id: string) {
  entities.value.delete(id)
  // Cleanup: remove from party if present
  const partyIdx = party.members.indexOf(id)
  if (partyIdx > -1) {
    party.members.splice(partyIdx, 1)
  }
}

function getEntity(id: string) {
  return entities.value.get(id)
}
```

## Party Actions

```ts
function addToParty(entityId: string) {
  if (!party.members.includes(entityId)) {
    party.members.push(entityId)
    if (!party.leader) {
      party.leader = entityId
    }
  }
}

function removeFromParty(entityId: string) {
  const idx = party.members.indexOf(entityId)
  if (idx > -1) {
    party.members.splice(idx, 1)
    if (party.leader === entityId) {
      party.leader = party.members[0] ?? null
    }
  }
}

function setPartyLeader(entityId: string) {
  if (party.members.includes(entityId)) {
    party.leader = entityId
  }
}
```

## Inventory Actions

```ts
export interface Item {
  id: string
  templateId: string
  quantity: number
}

function addToInventory(item: Item) {
  const existing = party.inventory.find(i => i.templateId === item.templateId)
  if (existing) {
    existing.quantity += item.quantity
  } else {
    party.inventory.push(item)
  }
}

function removeFromInventory(templateId: string, quantity = 1) {
  const idx = party.inventory.findIndex(i => i.templateId === templateId)
  if (idx > -1) {
    const item = party.inventory[idx]
    if (item) {
      item.quantity -= quantity
      if (item.quantity <= 0) {
        party.inventory.splice(idx, 1)
      }
    }
  }
}
```

## World Flags

Track persistent game state:

```ts
function setFlag(key: string, value: boolean | number) {
  worldFlags.value[key] = value
}

function getFlag(key: string): boolean | number | undefined {
  return worldFlags.value[key]
}

function hasFlag(key: string): boolean {
  return worldFlags.value[key] === true
}
```

Usage:
```ts
// Quest progress
gameStore.setFlag('quest_goblin_cleared', true)
gameStore.setFlag('goblin_kills', 5)

// Check state
if (gameStore.hasFlag('quest_goblin_cleared')) {
  // Show completion dialog
}
```

## Computed Getters

```ts
const partyEntities = computed(() =>
  party.members
    .map(id => entities.value.get(id))
    .filter((e): e is EntityState => !!e),
)

const partyCenter = computed((): Position => {
  const positions = partyEntities.value.map(e => e.position)
  if (!positions.length) return { x: 0, y: 0, z: 0 }
  return {
    x: positions.reduce((sum, p) => sum + p.x, 0) / positions.length,
    y: positions.reduce((sum, p) => sum + p.y, 0) / positions.length,
    z: positions.reduce((sum, p) => sum + p.z, 0) / positions.length,
  }
})

const hostileEntities = computed(() =>
  [...entities.value.values()].filter(e => e.hostile && e.hp && e.hp > 0),
)
```

## Equipment Actions

Equip and unequip weapons by slot:

```ts
function equipWeapon(entityId: string, weaponTemplateId: string, slot: 'mainHand' | 'offHand') {
  const entity = entities.value.get(entityId)
  if (entity) {
    entity.equipment = { ...entity.equipment, [slot]: weaponTemplateId }
  }
}

function unequipWeapon(entityId: string, slot: 'mainHand' | 'offHand') {
  const entity = entities.value.get(entityId)
  if (entity) {
    entity.equipment = { ...entity.equipment, [slot]: undefined }
  }
}
```

Usage:
```ts
gameStore.equipWeapon(heroId, 'shortsword', 'mainHand')
gameStore.unequipWeapon(heroId, 'mainHand')
```

See [Equipment](/characters/equipment) for how weapon models attach to character bones.

## Full Usage Example

```ts
const gameStore = useGameStore()

// Create party (once, persists across scenes)
const heroId = await gameStore.spawnFromTemplate('hero', { x: 0, y: 0, z: 0 })
gameStore.addToParty(heroId)
gameStore.selectEntity(heroId)
gameStore.equipWeapon(heroId, 'shortsword', 'mainHand')

// Load scene - spawns world entities, returns spawn point
const { spawnPoint } = await gameStore.loadScene('goblin_hideout')
gameStore.updateEntity(heroId, { position: spawnPoint })

// Combat: deal damage
const entity = gameStore.getEntity(heroId)
if (entity?.hp) {
  gameStore.updateEntity(heroId, { hp: entity.hp - 5 })
}

// Loot
gameStore.addToInventory({
  id: `potion_${Date.now()}`,
  templateId: 'healing_potion',
  quantity: 2,
})

// Track progress
gameStore.setFlag('first_combat_complete', true)

// Transition to new scene via exit
const { spawnPoint: newSpawn } = await gameStore.loadScene('forest_clearing', 'hideout_exit')
gameStore.updateEntity(heroId, { position: newSpawn })
```
