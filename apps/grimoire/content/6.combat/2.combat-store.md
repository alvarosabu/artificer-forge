---
title: Combat Store
description: Pinia store managing targeting state and attack execution
---

`useCombatStore` is a Pinia composition-API store that owns all combat state and orchestrates the attack sequence. It communicates with the 3D scene exclusively through `useSceneRefs`, keeping game logic decoupled from Vue component internals.

```ts
import { useCombatStore } from '~/stores/combat'

const combatStore = useCombatStore()
```

## State

| Property | Type | Default | Description |
|----------|------|---------|-------------|
| `isTargeting` | `Ref<boolean>` | `false` | Whether the player is picking a target |
| `isAttacking` | `Ref<boolean>` | `false` | Whether an attack sequence is in progress |
| `hoveredTargetId` | `Ref<string \| null>` | `null` | Entity ID of the enemy under the cursor |
| `cursorWorldPoint` | `Ref<Vector3 \| null>` | `null` | Current cursor position on the floor plane |

## Constants

```ts
const MELEE_RANGE = 1.5   // world units — attacker stops this far from target
const ATTACK_ANIM_MS = 800 // ms to wait after meleeAttack() before rolling damage
const HURT_ANIM_MS = 600   // ms to play hit animation before returning to idle
```

## Public API

### `enterTargeting()`

Puts the store into targeting mode. Adds a `cursor-crosshair` CSS class to `document.body` so the cursor changes globally.

```ts
function enterTargeting(): void
```

No-ops if `isAttacking` is `true` — prevents queuing attacks mid-sequence.

```ts
// Game.vue
onSlotActivated((slot) => {
  if (slot.id === 'attack-melee') combatStore.enterTargeting()
})
```

### `cancelTargeting()`

Resets all targeting state and removes the crosshair cursor. Called on Escape key or right-click from `Game.vue`.

```ts
function cancelTargeting(): void
```

```ts
defineShortcuts({ escape: () => combatStore.cancelTargeting() })

useEventListener('contextmenu', (e) => {
  if (combatStore.isTargeting) {
    e.preventDefault()
    combatStore.cancelTargeting()
  }
})
```

### `hoverTarget(entityId)`

Sets `hoveredTargetId` when the cursor enters a hostile entity during targeting. Passing `null` clears the hovered target.

```ts
function hoverTarget(entityId: string | null): void
```

Only hostile entities (where `entity.team === 'hostile'`) are accepted — neutral or party entities are ignored. Called from `Character.vue` pointer events.

### `confirmTarget(entityId)`

Validates that the entity is hostile, then cancels targeting and triggers `executeAttack`. This is the entry point that the player's click fires.

```ts
function confirmTarget(entityId: string): void
```

```ts
// Character.vue
function handleClick() {
  if (combatStore.isTargeting) {
    combatStore.confirmTarget(props.entityId)
  }
}
```

### `updateCursorPoint(point)`

Tracks the cursor's world-space position on the floor plane. The party leader's character rotates to face the cursor while targeting is active.

```ts
function updateCursorPoint(point: Vector3): void
```

Called from the invisible floor mesh in `Game.vue`:

```vue
<TresMesh @pointermove="(e) => combatStore.updateCursorPoint(e.point)">
  <TresPlaneGeometry :args="[100, 100]" />
  <TresMeshBasicMaterial :opacity="0" transparent />
</TresMesh>
```

## Internal: `executeAttack(targetId)`

Not exported from the store. Called automatically by `confirmTarget` after validation.

```ts
function executeAttack(targetId: string): void
```

Full sequence:

```
1. Guard: isAttacking already? → return
2. Look up attackerId = gameStore.party.leader
3. Resolve attackerRef and targetRef from useSceneRefs
4. Read attacker/target positions from gameStore
5. Compute meleePos: target position + (attacker→target direction × MELEE_RANGE)
6. attackerRef.moveTo(meleePos)
7. attackerRef.onArrive(() => {
     attackerRef.meleeAttack()
     setTimeout(ATTACK_ANIM_MS, () => {
       const { value, critical } = rollDamage(targetEntity.stats)
       gameStore.updateEntity(targetId, { hp: newHp })
       targetRef.showDamage(value, 'physical', critical)
       targetRef.play(AnimationName.HIT_A, 0.2)
       setTimeout(HURT_ANIM_MS, () => {
         targetRef.play(AnimationName.IDLE_A, 0.3)
         attackerRef.play(AnimationName.IDLE_A, 0.3)
         isAttacking = false
       })
     })
   })
```

The melee position calculation ensures the attacker always stops `MELEE_RANGE` units in front of the target, facing it:

```ts
const dir = aVec.clone().sub(tVec).normalize()   // attacker → target direction
const meleePos = tVec.clone().add(dir.multiplyScalar(MELEE_RANGE))
meleePos.y = 0  // keep on the floor
```

## `rollDamage(stats)`

Module-level helper (not exported). Rolls a d6, adds the strength modifier, doubles on a critical.

```ts
function rollDamage(stats: Record<string, number> = {}): { value: number, critical: boolean }
```

| Roll | Result |
|------|--------|
| 1–5 | `d6 + strMod`, minimum 1 |
| 6 (natural) | `(d6 × 2) + strMod` — critical |

```ts
const strMod = Math.floor(((stats.strength ?? 10) - 10) / 2)
```

::alert{type="warning"}
`executeAttack` currently passes `targetEntity.stats` to `rollDamage`, which means damage is calculated from the **target's** strength, not the attacker's. This is a known bug — it should use the attacker's stats. Fix by resolving `attackerEntity.stats` before the `onArrive` callback.
::

## Full Store Source

```ts
import { Vector3 } from 'three'
import { AnimationName, useSceneRefs } from '@artificer-forge/composables'

const MELEE_RANGE = 1.5
const ATTACK_ANIM_MS = 800
const HURT_ANIM_MS = 600

function rollDamage(stats: Record<string, number> = {}): { value: number, critical: boolean } {
  const d6 = Math.floor(Math.random() * 6) + 1
  const strMod = Math.floor(((stats.strength ?? 10) - 10) / 2)
  const critical = d6 === 6
  const value = critical ? d6 * 2 + strMod : d6 + strMod
  return { value: Math.max(1, value), critical }
}

export const useCombatStore = defineStore('combat', () => {
  const gameStore = useGameStore()
  const { getCharacterRef } = useSceneRefs()

  const isTargeting = ref(false)
  const isAttacking = ref(false)
  const hoveredTargetId = ref<string | null>(null)
  const cursorWorldPoint = ref<Vector3 | null>(null)

  function enterTargeting() {
    if (isAttacking.value) return
    isTargeting.value = true
    document.body.classList.add('cursor-crosshair')
  }

  function cancelTargeting() {
    isTargeting.value = false
    hoveredTargetId.value = null
    cursorWorldPoint.value = null
    document.body.classList.remove('cursor-crosshair')
  }

  function updateCursorPoint(point: Vector3) {
    if (!isTargeting.value) return
    cursorWorldPoint.value = point
    const leaderId = gameStore.party.leader
    if (!leaderId) return
    getCharacterRef(leaderId)?.lookAt(point)
  }

  function hoverTarget(entityId: string | null) {
    if (!isTargeting.value) return
    if (!entityId) { hoveredTargetId.value = null; return }
    const entity = gameStore.getEntity(entityId)
    hoveredTargetId.value = entity?.team === 'hostile' ? entityId : null
  }

  function confirmTarget(entityId: string) {
    if (!isTargeting.value) return
    const entity = gameStore.getEntity(entityId)
    if (entity?.team !== 'hostile') return
    cancelTargeting()
    executeAttack(entityId)
  }

  function executeAttack(targetId: string) {
    if (isAttacking.value) return
    const attackerId = gameStore.party.leader
    if (!attackerId) return

    const attackerRef = getCharacterRef(attackerId)
    const targetRef = getCharacterRef(targetId)
    if (!attackerRef || !targetRef) return

    const attackerPos = gameStore.getEntity(attackerId)?.position
    const targetPos = gameStore.getEntity(targetId)?.position
    if (!attackerPos || !targetPos) return

    isAttacking.value = true

    const aVec = new Vector3(attackerPos.x, attackerPos.y, attackerPos.z)
    const tVec = new Vector3(targetPos.x, targetPos.y, targetPos.z)
    const dir = aVec.clone().sub(tVec).normalize()
    const meleePos = tVec.clone().add(dir.multiplyScalar(MELEE_RANGE))
    meleePos.y = 0

    attackerRef.moveTo(meleePos)

    let arrived = false
    attackerRef.onArrive(() => {
      if (arrived) return
      arrived = true

      attackerRef.meleeAttack()

      setTimeout(() => {
        const targetEntity = gameStore.getEntity(targetId)
        if (!targetEntity) { isAttacking.value = false; return }

        const { value, critical } = rollDamage(targetEntity.stats ?? {})
        const newHp = Math.max(0, (targetEntity.hp ?? 0) - value)

        gameStore.updateEntity(targetId, { hp: newHp })
        targetRef.showDamage(value, 'physical', critical)
        targetRef.play(AnimationName.HIT_A, 0.2)

        setTimeout(() => {
          targetRef.play(AnimationName.IDLE_A, 0.3)
          attackerRef.play(AnimationName.IDLE_A, 0.3)
          isAttacking.value = false
        }, HURT_ANIM_MS)
      }, ATTACK_ANIM_MS)
    })
  }

  return {
    isTargeting,
    isAttacking,
    hoveredTargetId,
    enterTargeting,
    cancelTargeting,
    hoverTarget,
    confirmTarget,
    updateCursorPoint,
    cursorWorldPoint,
  }
})
```

## Usage in Game.vue

```ts
const { onSlotActivated } = useActionBar()
const combatStore = useCombatStore()

// Enter targeting when the melee slot is activated
onSlotActivated((slot) => {
  if (slot.id === 'attack-melee') combatStore.enterTargeting()
})

// Cancel on Escape
defineShortcuts({ escape: () => combatStore.cancelTargeting() })

// Show reticle at hovered target's feet
const reticlePosition = computed<[number, number, number]>(() => {
  const id = combatStore.hoveredTargetId
  if (!id) return [0, 0.01, 0]
  const entity = gameStore.getEntity(id)
  if (!entity) return [0, 0.01, 0]
  return [entity.position.x, 0.01, entity.position.z]
})
```
