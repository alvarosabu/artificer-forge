# Entity Templates

Templates are YAML files in `content/entities/` that define base entity data. Nuxt Content loads and validates them at build time.

## Configuration

First, configure the collection in `content.config.ts`:

```ts
import { defineContentConfig, defineCollection, z } from '@nuxt/content'

export default defineContentConfig({
  collections: {
    entities: defineCollection({
      type: 'data',
      source: 'entities/**/*.yaml',
      schema: z.object({
        templateId: z.string(),
        type: z.enum(['character', 'item', 'interactable']),
        subtype: z.string().optional(),
        name: z.string(),

        // Character fields
        faction: z.string().optional(),
        hostile: z.boolean().optional(),
        controllable: z.boolean().optional(),
        stats: z.record(z.number()).optional(),
        ai: z.object({
          behavior: z.string(),
        }).passthrough().optional(),
        model: z.string().optional(),
        animations: z.record(z.string()).optional(),

        // Item fields
        damage: z.object({
          dice: z.string(),
          type: z.string(),
        }).optional(),
        properties: z.array(z.string()).optional(),
        effect: z.object({
          type: z.string(),
          dice: z.string().optional(),
          bonus: z.number().optional(),
        }).optional(),
        weight: z.number().optional(),
        value: z.number().optional(),
        usable: z.boolean().optional(),
        stackable: z.boolean().optional(),

        // Interactable fields
        locked: z.boolean().optional(),
        lockDifficulty: z.number().optional(),
        hp: z.number().optional(),
        destructible: z.boolean().optional(),
        lootTable: z.array(z.object({
          id: z.string(),
          quantity: z.union([z.number(), z.array(z.number())]).optional(),
          chance: z.number(),
        })).optional(),
      }),
    }),
  },
})
```

## Character Templates

### Companion (Controllable)

```yaml
# content/entities/characters/companions/ranger.yaml
templateId: ranger
type: character
name: Elara the Ranger

faction: party
hostile: false
controllable: true

stats:
  hp: 28
  maxHp: 28
  ac: 14
  str: 12
  dex: 16
  con: 14
  int: 10
  wis: 14
  cha: 10

model: /models/Ranger.glb
animations:
  idle: Idle_Neutral_A
  walk: Walk_Forward_A
  attack: Melee_1H_Attack_Chop
  death: Death_A
```

### Enemy (AI-Controlled)

```yaml
# content/entities/characters/enemies/goblin-scout.yaml
templateId: goblin_scout
type: character
name: Goblin Scout

faction: enemies
hostile: true
controllable: false

stats:
  hp: 7
  maxHp: 7
  ac: 13
  str: 8
  dex: 14
  con: 10
  int: 10
  wis: 8
  cha: 8

ai:
  behavior: aggressive
  aggroRange: 10
  preferredRange: melee

model: /models/Goblin.glb
animations:
  idle: Idle_Combat
  walk: Walk_Forward_A
  attack: Melee_1H_Attack_Stab
  death: Death_A
```

## Item Templates

### Weapon

```yaml
# content/entities/items/longsword.yaml
templateId: longsword
type: item
subtype: weapon
name: Longsword

damage:
  dice: 1d8
  type: slashing

properties:
  - versatile

weight: 3
value: 15
```

### Consumable

```yaml
# content/entities/items/healing-potion.yaml
templateId: healing_potion
type: item
subtype: consumable
name: Healing Potion

effect:
  type: heal
  dice: 2d4
  bonus: 2

value: 50
usable: true
stackable: true
maxStack: 10
```

## Interactable Templates

```yaml
# content/entities/interactables/chest-common.yaml
templateId: chest_common
type: interactable
subtype: container
name: Wooden Chest

locked: false
lockDifficulty: 0

hp: 20
destructible: true

model: /models/chest.glb
animations:
  closed: Closed
  open: Open

lootTable:
  - id: gold
    quantity: [5, 20]  # Random range
    chance: 0.9
  - id: healing_potion
    chance: 0.3
  - id: longsword
    chance: 0.1
```

## Querying Templates

Use `useEntityTemplates` composable for typed queries:

```ts
// composables/useEntityTemplates.ts
export function useEntityTemplates() {
  async function getAll() {
    return queryCollection('entities').all()
  }

  async function getCharacters() {
    return queryCollection('entities')
      .where('type', '=', 'character')
      .all()
  }

  async function getEnemies() {
    return queryCollection('entities')
      .where('type', '=', 'character')
      .where('hostile', '=', true)
      .all()
  }

  async function getCompanions() {
    return queryCollection('entities')
      .where('type', '=', 'character')
      .where('controllable', '=', true)
      .where('hostile', '=', false)
      .all()
  }

  async function getItems() {
    return queryCollection('entities')
      .where('type', '=', 'item')
      .all()
  }

  async function getTemplateById(templateId: string) {
    return queryCollection('entities')
      .where('templateId', '=', templateId)
      .first()
  }

  return {
    getAll,
    getCharacters,
    getEnemies,
    getCompanions,
    getItems,
    getTemplateById,
  }
}
```

## Directory Structure

Organize templates by type and subtype:

```
content/entities/
├── characters/
│   ├── companions/
│   │   ├── ranger.yaml
│   │   └── player.yaml
│   └── enemies/
│       ├── goblin-scout.yaml
│       └── goblin-warrior.yaml
├── items/
│   ├── weapons/
│   │   ├── longsword.yaml
│   │   └── shortbow.yaml
│   └── consumables/
│       └── healing-potion.yaml
└── interactables/
    ├── chest-common.yaml
    └── chest-locked.yaml
```

::alert{type="info"}
After adding or modifying YAML files, restart the dev server to rebuild the content index.
::
