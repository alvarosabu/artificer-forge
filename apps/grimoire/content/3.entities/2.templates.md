# Entity Templates

Templates are YAML files in `content/entities/` that define **what** an entity is - its model, stats, default values. [Scenes](/entities/scenes) then reference templates to define **where** entities are placed in the world.

::alert{type="info"}
Templates define blueprints. Scenes define placements. Runtime state lives in the [Game Store](/core-concepts/game-store).
::

## Configuration

First, configure the collection in `content.config.ts`:

```ts
import { defineContentConfig, defineCollection, z } from '@nuxt/content'

export default defineContentConfig({
  collections: {
    entities: defineCollection({
      type: 'data',
      source: 'entities/**/*.yaml',
      schema: z.object({
        templateId: z.string(),
        type: z.enum(['character', 'item', 'interactable']),
        subtype: z.string().optional(),
        name: z.string(),

        // Character fields
        class: z.string().optional(),
        level: z.number().optional(),
        race: z.string().optional(),
        faction: z.string().optional(),
        team: z.enum(['player', 'ally', 'neutral', 'hostile']).optional(),
        controllable: z.boolean().optional(),
        stats: z.record(z.number()).optional(),
        ai: z.object({
          behavior: z.string(),
        }).passthrough().optional(),
        model: z.string().optional(),
        animations: z.record(z.string()).optional(),

        // Item fields
        damage: z.object({
          dice: z.string(),
          type: z.string(),
        }).optional(),
        properties: z.array(z.string()).optional(),
        effect: z.object({
          type: z.string(),
          dice: z.string().optional(),
          bonus: z.number().optional(),
        }).optional(),
        weight: z.number().optional(),
        value: z.number().optional(),
        usable: z.boolean().optional(),
        stackable: z.boolean().optional(),

        // Equipment (starting gear for characters)
        equipment: z.object({
          mainHand: z.string().optional(),
          offHand: z.string().optional(),
        }).optional(),

        // Interactable fields
        locked: z.boolean().optional(),
        lockDifficulty: z.number().optional(),
        hp: z.number().optional(),
        destructible: z.boolean().optional(),
        lootTable: z.array(z.object({
          id: z.string(),
          quantity: z.union([z.number(), z.array(z.number())]).optional(),
          chance: z.number(),
        })).optional(),
      }),
    }),
  },
})
```

## Character Templates

### Companion (Controllable)

```yaml
# content/entities/characters/companions/hero.yaml
templateId: hero
type: character
name: Hero
class: fighter
level: 1
race: Human

team: player
controllable: true

stats:
  hp: 38
  maxHp: 38
  ac: 14
  strength: 12
  dexterity: 16
  constitution: 12
  intelligence: 10
  wisdom: 14
  charisma: 10
  maxActions: 1
  maxBonusActions: 1

model: /models/characters/hero.glb
animations:
  idle: Idle
  walk: Walk
  run: Run
  attack: Bow_Attack
  dodge: Dodge
```

### Enemy (AI-Controlled)

```yaml
# content/entities/characters/enemies/orc_scout.yaml
templateId: orc_scout
type: character
subtype: enemy
name: Orc Scout
level: 1
race: Orc

faction: orc
team: hostile
controllable: false

stats:
  hp: 12
  maxHp: 12
  ac: 13
  strength: 8
  dexterity: 14
  constitution: 10
  intelligence: 8
  wisdom: 8
  charisma: 6

ai:
  behavior: aggressive
  alertRange: 10
  attackRange: 5
  fleeThreshold: 0.2

model: /models/characters/orc.glb
animations:
  idle: Idle
  walk: Walk
  run: Run
  attack: Slash

equipment:
  mainHand: orc_axe
```

## Item Templates

### Weapon

Weapons need a `model` field for bone attachment via [useEquipment](/characters/equipment):

```yaml
# content/entities/items/weapons/shortsword.yaml
templateId: shortsword
type: item
subtype: weapon
name: Shortsword
model: /models/items/weapons/Sword_1Handed.glb

damage:
  dice: "1d8"
  type: slashing

properties:
  - versatile

weight: 3
value: 15
```

### Consumable

```yaml
# content/entities/items/healing-potion.yaml
templateId: healing_potion
type: item
subtype: consumable
name: Healing Potion

effect:
  type: heal
  dice: 2d4
  bonus: 2

value: 50
usable: true
stackable: true
maxStack: 10
```

## Interactable Templates

```yaml
# content/entities/interactables/chest-common.yaml
templateId: chest_common
type: interactable
subtype: container
name: Wooden Chest

locked: false
lockDifficulty: 0

hp: 20
destructible: true

model: /models/chest.glb
animations:
  closed: Closed
  open: Open

lootTable:
  - id: gold
    quantity: [5, 20]  # Random range
    chance: 0.9
  - id: healing_potion
    chance: 0.3
  - id: longsword
    chance: 0.1
```

## Querying Templates

Use `useEntityTemplates` composable for typed queries:

```ts
// composables/useEntityTemplates.ts
export function useEntityTemplates() {
  async function getAll() {
    return queryCollection('entities').all()
  }

  async function getCharacters() {
    return queryCollection('entities')
      .where('type', '=', 'character')
      .all()
  }

  async function getEnemies() {
    return queryCollection('entities')
      .where('type', '=', 'character')
      .where('hostile', '=', true)
      .all()
  }

  async function getCompanions() {
    return queryCollection('entities')
      .where('type', '=', 'character')
      .where('controllable', '=', true)
      .where('hostile', '=', false)
      .all()
  }

  async function getItems() {
    return queryCollection('entities')
      .where('type', '=', 'item')
      .all()
  }

  async function getTemplateById(templateId: string) {
    return queryCollection('entities')
      .where('templateId', '=', templateId)
      .first()
  }

  return {
    getAll,
    getCharacters,
    getEnemies,
    getCompanions,
    getItems,
    getTemplateById,
  }
}
```

## Directory Structure

Organize templates by type and subtype:

```
content/entities/
├── characters/
│   ├── companions/
│   │   └── hero.yaml
│   ├── enemies/
│   │   └── orc_scout.yaml
│   └── npcs/
│       └── guard-captain.yaml
├── items/
│   └── weapons/
│       └── orc_axe.yaml
└── interactables/
    └── chest-common.yaml
```

::alert{type="info"}
After adding or modifying YAML files, restart the dev server to rebuild the content index.
::
