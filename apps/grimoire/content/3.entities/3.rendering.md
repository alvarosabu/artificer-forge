# Rendering Entities

The experience page (smart) manages entity lifecycle and passes data to Character components (dumb) for rendering.

## Pattern: Smart vs Dumb

| Component | Responsibilities |
|-----------|-----------------|
| **Experience** (smart) | Access store, spawn entities, compute filtered lists |
| **Character** (dumb) | Receive props, load model, play animations |

This keeps Character reusable and testable - it doesn't need store access.

## Experience Page

```vue
<!-- pages/model/experience.vue -->
<script setup lang="ts">
import { Floor } from '@artificer-forge/components'

const gameStore = useGameStore()

// Track spawned entity for cleanup
const heroEntityId = ref<string | null>(null)

onMounted(async () => {
  // Spawn hero at origin
  const entityId = await gameStore.spawnFromTemplate('hero', { x: 0, y: 0, z: 0 })
  heroEntityId.value = entityId
  gameStore.addToParty(entityId)
})

// Filter entities by type for rendering
const characterEntities = computed(() => {
  return [...gameStore.entities.values()].filter(e => e.type === 'character')
})
</script>

<template>
  <TresFog :args="['#020420', 10, 30]" />
  <TresPerspectiveCamera :position="[5, 5, 5]" :near="0.1" :far="100" />
  <OrbitControls />

  <TresAmbientLight :intensity="0.8" />
  <TresDirectionalLight
    :position="[5, 5, 5]"
    :intensity="1.5"
    cast-shadow
  />

  <!-- Render character entities from store -->
  <Suspense>
    <Character
      v-for="entity in characterEntities"
      :key="entity.id"
      :position="[entity.position.x, entity.position.y, entity.position.z]"
      :model="entity.model"
    />
  </Suspense>

  <TresAxesHelper />
  <Floor />
</template>
```

## Character Component (Dumb)

```vue
<!-- components/Character.vue -->
<script setup lang="ts">
import { useGLTF } from '@tresjs/cientos'
import { useCharacterAnimations, AnimationName } from '~/composables/useCharacterAnimations'

const props = withDefaults(defineProps<{
  model?: string
  position?: [number, number, number]
  rotation?: [number, number, number]
}>(), {
  model: '/models/characters/hero.glb',
  position: () => [0, 0, 0],
  rotation: () => [0, 0, 0],
})

const { nodes } = useGLTF(props.model)
const rig = computed(() => nodes.value?.Rig_Medium)

const { actions, currentAnimName, play, stop } = useCharacterAnimations(rig)

// Expose animation controls to parent
defineExpose({ play, stop, currentAnimName, actions, AnimationName })
</script>

<template>
  <TresGroup :position="props.position" :rotation="props.rotation">
    <primitive v-if="rig" :object="rig" cast-shadow receive-shadow />
  </TresGroup>
</template>
```

## Spawning Multiple Entities

```ts
onMounted(async () => {
  // Spawn party
  const heroId = await gameStore.spawnFromTemplate('hero', { x: 0, y: 0, z: 0 })
  const playerId = await gameStore.spawnFromTemplate('player', { x: -1, y: 0, z: 1 })

  gameStore.addToParty(heroId)
  gameStore.addToParty(playerId)

  // Spawn enemies
  await gameStore.spawnFromTemplate('goblin_scout', { x: 5, y: 0, z: 0 })
  await gameStore.spawnFromTemplate('goblin_scout', { x: 6, y: 0, z: 2 })
  await gameStore.spawnFromTemplate('goblin_warrior', { x: 5.5, y: 0, z: 1 })
})
```

## Filtered Rendering

Render different entity types with different components:

```vue
<script setup>
const characterEntities = computed(() =>
  [...gameStore.entities.values()].filter(e => e.type === 'character')
)

const itemEntities = computed(() =>
  [...gameStore.entities.values()].filter(e => e.type === 'item')
)

const interactableEntities = computed(() =>
  [...gameStore.entities.values()].filter(e => e.type === 'interactable')
)
</script>

<template>
  <!-- Characters with animation support -->
  <Suspense>
    <Character
      v-for="entity in characterEntities"
      :key="entity.id"
      :position="[entity.position.x, entity.position.y, entity.position.z]"
      :model="entity.model"
    />
  </Suspense>

  <!-- Items (simple models) -->
  <Suspense>
    <ItemModel
      v-for="entity in itemEntities"
      :key="entity.id"
      :position="[entity.position.x, entity.position.y, entity.position.z]"
      :model="entity.model"
    />
  </Suspense>

  <!-- Interactables (chests, doors) -->
  <Suspense>
    <Interactable
      v-for="entity in interactableEntities"
      :key="entity.id"
      :position="[entity.position.x, entity.position.y, entity.position.z]"
      :model="entity.model"
      :opened="entity.opened"
    />
  </Suspense>
</template>
```

## Reactive Updates

When store state changes, the scene updates automatically:

```ts
// Move an entity - scene re-renders
gameStore.updateEntity(entityId, {
  position: { x: newX, y: 0, z: newZ }
})

// Remove entity - disappears from scene
gameStore.removeEntity(entityId)

// Update HP - could trigger death animation
const entity = gameStore.getEntity(entityId)
if (entity && entity.hp !== undefined) {
  const newHp = entity.hp - damage
  gameStore.updateEntity(entityId, { hp: newHp })

  if (newHp <= 0) {
    // Trigger death, remove after animation
    characterRef.value?.play('death')
    setTimeout(() => gameStore.removeEntity(entityId), 2000)
  }
}
```

## Entity References for Animation Control

To control individual character animations, use template refs with the entity ID:

```vue
<script setup>
const characterRefs = ref<Record<string, any>>({})

function setCharacterRef(entityId: string, ref: any) {
  if (ref) characterRefs.value[entityId] = ref
}

function playAnimation(entityId: string, anim: string) {
  characterRefs.value[entityId]?.play(anim)
}
</script>

<template>
  <Suspense>
    <Character
      v-for="entity in characterEntities"
      :key="entity.id"
      :ref="(ref) => setCharacterRef(entity.id, ref)"
      :position="[entity.position.x, entity.position.y, entity.position.z]"
      :model="entity.model"
    />
  </Suspense>
</template>
```

::alert{type="info"}
Wrap async model loading components in `<Suspense>` to handle loading states gracefully.
::
